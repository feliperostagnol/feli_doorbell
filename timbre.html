export let currentToken = null; export let validUntil = 0; export default function handler(req, res) { const now = Date.now(); if (!currentToken || now > validUntil) { currentToken = Math.random().toString(36).substring(2) + Math.random().toString(36).substring(2); validUntil = now + 4 * 60 * 60 * 1000; // 4h } res.writeHead(302, { Location: `/timbre.html?token=${currentToken}` }); res.end(); }
